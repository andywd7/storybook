import {
  Meta,
  Title,
  Subtitle,
  Story,
  Canvas,
  ArgsTable 
} from '@storybook/addon-docs/blocks'
import { Var } from "@dan-nz/storybook-documentation-primitives"
import groupBy from "lodash/groupBy"
import { store } from "../../components/store"
import { SPACING_08 } from "../../tokens/web/js/tokensAndPalette"

export const colors = store.tokens.color.filter((item) => item.attributes.item)
export const backWhite = store.tokens.color.filter((item) => !item.attributes.item)

export const ccc = () => groupBy(colors, "attributes.type")

export const tokensColors = Object.values(ccc()).map((t, i) => {
  // console.log(t[i].value)
  const styles = {
    // backgroundColor: t[2].value,
    borderRadius: SPACING_08,
    height: SPACING_08,
    width: SPACING_08
  }
  return (
    <div key={i}>
      <h3>{t[0].attributes.type.toString()}</h3>
      <table className="table">
        <thead className="thead-dark">
          <tr>
              <th width="120">Token</th>
              <th width="120">Value</th>
          </tr>
        </thead>
        <tbody>
          {
            Object.values(t).map((val, index) => {
              return (
                <tr key={index}>
                  <td><Var>{"$" + val.name}</Var></td>
                  <td>
                    <div className="token-value">
                      <div>
                        {val.name}
                        <br/>
                        -
                        <br/>
                        <Var>{val.value}</Var>
                      </div>
                      <div style={styles} style={{backgroundColor: val.value}} />
                    </div>
                  </td>
                </tr>
              )
            })
          }
        </tbody>
      </table>
    </div>
  )
})

<Meta title='TOKENS/Colors/Col' parameters={{ previewTabs: { canvas: { hidden: true } } }} />

<Title />

<Subtitle>Tokens subtitle</Subtitle>

<div>{tokensColors}</div>
